const AnimatedText = ({ text, className = "" }) => {
  return (
    <div className={`flex flex-wrap justify-center ${className}`}>
      {[...text].map((char, i) => (
        <span
          key={i}
          className="inline-block transition-all duration-300 hover:scale-110 hover:text-gray-900"
        >
          {char === " " ? "\u00A0" : char}
        </span>
      ))}
    </div>
  );
};
